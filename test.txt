    <h2>Select a Folder to Generate Regular Expression for File Filtering</h2>

    <!-- Folder input -->
    <input type="file" id="folderInput" webkitdirectory directory multiple />
    
    <h3>Generated Regular Expression:</h3>
    <textarea id="regexOutput" rows="4" cols="50"></textarea>

    <script>
        // Function to generate regex based on file names
        document.getElementById('folderInput').addEventListener('change', function(event) {
            let files = event.target.files;
            let regexParts = [];
            
            for (let i = 0; i < files.length; i++) {
                let fileName = files[i].name;
                // Escape special regex characters in file names
                fileName = fileName.replace(/[-\/\\^$*+?.()|[\]{}]/g, '\\$&');
                regexParts.push(fileName);
            }

            // Join file names with | (or) to create a regex pattern
            let regex = regexParts.join('|');
            
            // Display the regular expression in the textarea
            document.getElementById('regexOutput').value = regex;
        });
    </script>
