    <script>
        function parseFieldList(fieldListXML) {
            const parser = new DOMParser();
            const xmlDoc = parser.parseFromString(fieldListXML, "text/xml");
            const fields = xmlDoc.getElementsByTagName("FIELD");
            const fieldMap = {};

            for (let i = 0; i < fields.length; i++) {
                const field = fields[i];
                const fieldName = field.textContent.trim();
                fieldMap[fieldName] = {
                    DESC: field.getAttribute("DESC"),
                    DATATYPE: field.getAttribute("DATATYPE"),
                    EEType: field.getAttribute("EEType")
                };
            }

            return fieldMap;
        }

        function convertToXML() {
            const sqlInput = document.getElementById("sqlInput").value;
            const fieldListInput = document.getElementById("fieldListInput").value;

            const fieldMap = parseFieldList(fieldListInput);
            let xmlOutput = "<CLAUSE>\n";

            const tokens = sqlInput.split(/\s+/);
            let index = 0;

            while (index < tokens.length) {
                const token = tokens[index];

                if (token.match(/AND|OR|=|<>|>|<|IN|NOT/)) {
                    xmlOutput += `    <SQL TYPE="SQL"> ${token} </SQL>\n`;
                } else if (token.match(/ORDER\s+BY/i)) {
                    xmlOutput += `    <SQL TYPE="SQL"> ${token} </SQL>\n`;
                    index++;
                    const fieldInfo = fieldMap[tokens[index]];
                    if (fieldInfo) {
                        xmlOutput += `    <SQL DESC="${fieldInfo.DESC}" DATATYPE="${fieldInfo.DATATYPE}" EEType="${fieldInfo.EEType}" TYPE="FLD">${tokens[index]}</SQL>\n`;
                    }
                } else if (token.toLowerCase() === 'operand') {
                    xmlOutput += `    <SQL TYPE="OP"> ${token} </SQL>\n`;
                    index++;
                } else if (token.match(/^'.*'$/)) { // Matches 'value'
                    xmlOutput += `    <SQL TYPE="SQL"> '${token.slice(1)} </SQL>\n`; // Opening quote
                    xmlOutput += `    <SQL TYPE="OP"> ${token.slice(1, -1)} </SQL>\n`; // Value
                    xmlOutput += `    <SQL TYPE="SQL"> ' </SQL>\n`; // Closing quote
                } else if (fieldMap[token]) {
                    const fieldInfo = fieldMap[token];
                    xmlOutput += `    <SQL DESC="${fieldInfo.DESC}" DATATYPE="${fieldInfo.DATATYPE}" EEType="${fieldInfo.EEType}" TYPE="FLD">${token}</SQL>\n`;
                } else {
                    xmlOutput += `    <SQL TYPE="SQL"> ${token} </SQL>\n`;
                }
                index++;
            }

            xmlOutput += "</CLAUSE>";
            document.getElementById("xmlOutput").textContent = xmlOutput;
        }
    </script>
