function checkMandatoryFields(divId) {
    // Get the parent div using the divId
    var parentDiv = document.getElementById(divId);
    
    // Flag to check if all mandatory fields are filled
    var allFilled = true;
    
    // Loop through each mandatory field (_M) within the div
    $(parentDiv).find('input[class$="_M"], select[class$="_M"]').each(function() {
        if ($(this).val() === '' || $(this).val() === null) {
            allFilled = false;
        }
    });
    
    // If all fields are filled, show the tick; otherwise, hide it
    if (allFilled) {
        $(parentDiv).find('img[id^="tick"]').show();
    } else {
        $(parentDiv).find('img[id^="tick"]').hide();
    }
}
