function disableAllFields(divId) {
    // Get the div by its ID
    var div = document.getElementById(divId);
    
    // Find all input, textarea, select, and button elements inside the div
    var fields = div.querySelectorAll('input, textarea, select, button');
    
    // Loop through all the elements and disable them
    fields.forEach(function(field) {
        field.disabled = true;
    });
}

// Call the function and pass the 'outwardMainDiv' id
disableAllFields('outwardMainDiv');
