function parseFieldList(fieldListInput) {
    const lines = fieldListInput.split('\n');
    const fieldMap = {};

    for (let line of lines) {
        // Trim the line to remove any leading/trailing whitespace
        line = line.trim();

        // Skip empty lines
        if (line === '') continue;

        const parts = line.split(',');

        // Ensure that the line has at least three elements (fieldName, desc, dt)
        if (parts.length < 3) {
            console.warn(`Skipping invalid field list line: ${line}`);
            continue;
        }

        const [fieldName, desc, dt, eeType] = parts.map(part => part.trim());

        fieldMap[fieldName] = { 
            DESC: desc, 
            DT: dt, 
            EEType: eeType || "" // Use an empty string if eeType is undefined
        };
    }

    return fieldMap;
}
