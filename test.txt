@WebServlet("/downloadImage")
public class ImageDownloadServlet extends HttpServlet {
    
    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        String c_img_indx = request.getParameter("c_img_indx");
        String c_main_ref = request.getParameter("c_main_ref");

        if (c_img_indx == null || c_main_ref == null) {
            response.sendError(HttpServletResponse.SC_BAD_REQUEST, "Missing parameters.");
            return;
        }

        // Retrieve the base64 image string from the database
        String base64Image = getImageFromDatabase(c_img_indx, c_main_ref);

        if (base64Image == null) {
            response.sendError(HttpServletResponse.SC_NOT_FOUND, "Image not found.");
            return;
        }

        // Decode the base64 image
        byte[] imageBytes = Base64.getDecoder().decode(base64Image);

        // Set the content type and disposition (you can adjust the content type based on the image format)
        response.setContentType("image/jpeg"); // Change to image/png or other type if needed
        response.setHeader("Content-Disposition", "attachment; filename=image_" + c_img_indx + ".jpg");

        // Write the image bytes to the response output stream
        response.setContentLength(imageBytes.length);
        ServletOutputStream outStream = response.getOutputStream();
        outStream.write(imageBytes);
        outStream.flush();
        outStream.close();
    }

    private String getImageFromDatabase(String c_img_indx, String c_main_ref) {
        String base64Image = null;
        Connection conn = null;
        PreparedStatement stmt = null;
        ResultSet rs = null;

        try {
            // Get connection to the database
            conn = DatabaseConnection.getConnection(); // Ensure DatabaseConnection is implemented

            // SQL Query to retrieve the base64 image based on c_img_indx and c_main_ref
            String sql = "SELECT image_data FROM trx_image_dtal WHERE c_img_indx = ? AND c_main_ref = ?";
            stmt = conn.prepareStatement(sql);
            stmt.setString(1, c_img_indx);
            stmt.setString(2, c_main_ref);

            rs = stmt.executeQuery();

            if (rs.next()) {
                base64Image = rs.getString("image_data"); // Adjust column name if different
            }

        } catch (SQLException e) {
            e.printStackTrace();
        } finally {
            try { if (rs != null) rs.close(); } catch (SQLException e) { e.printStackTrace(); }
            try { if (stmt != null) stmt.close(); } catch (SQLException e) { e.printStackTrace(); }
            try { if (conn != null) conn.close(); } catch (SQLException e) { e.printStackTrace(); }
        }

        return base64Image;
    }
}
