import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class TagExtractor {

    public static void main(String[] args) {
        String filePath = "path/to/your/input/file.txt";  // Update this with the actual file path

        try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
            String line;
            // Pattern to match id, title, and class attributes independently
            Pattern pattern = Pattern.compile("(?<=id=\")([^\"]+)|(?<=title=\")([^\"]+)|(?<=class=\")([^\"]+)");

            while ((line = br.readLine()) != null) {
                System.out.println("Reading line: " + line);  // Debug print
                Matcher matcher = pattern.matcher(line);

                String id = "N/A";
                String title = "N/A";
                String className = "N/A";

                while (matcher.find()) {
                    if (line.contains("id=\"")) {
                        id = matcher.group(1) != null ? matcher.group(1) : id;
                    }
                    if (line.contains("title=\"")) {
                        title = matcher.group(2) != null ? matcher.group(2) : title;
                    }
                    if (line.contains("class=\"")) {
                        className = matcher.group(3) != null ? matcher.group(3) : className;
                    }
                }

                // Print the extracted values
                System.out.println(title + " - " + id + " - " + className);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
